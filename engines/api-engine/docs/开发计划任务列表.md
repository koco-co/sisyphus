```python
class RetryManager:
    def execute_with_retry(self, func, max_retries: int, delay: int):
        """带重试的执行"""

        for attempt in range(1, max_retries + 1):
            try:
                return func()
            except Exception as e:
                if attempt == max_retries:
                    raise
                time.sleep(delay / 1000)
```
